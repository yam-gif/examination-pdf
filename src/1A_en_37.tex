\subsection*{1A~演習問題第37問}
\begin{enumerate}[left=0pt,label=(\arabic*)]
    \item 
        $x+2y=1$のとき,\:$x^2+y^2$の最小値を求めよ。
    \item 
        $x^2+2y^2=1$のとき,\:$x^2+4y$の最大値,\:最小値を求めよ。
    \item 
        $y=-(x^2-4x+1)^2+2x^2-8x-1~(0\leqq x \leqq 3)$について
        \begin{enumerate}[left=0pt,label=(\roman*)]
            \item
                $x^2-4x+1=t$とおくとき,\:$t$のとりうる値の範囲を求めよ。
            \item 
                $y$の最大値,\:最小値を求めよ。
        \end{enumerate}
\end{enumerate}
